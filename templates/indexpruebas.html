function eliminarVendedor(cedula) { var vendedores = JSON.parse(localStorage.getItem('Vendedores')) || []; var vendedor = vendedores.find(v => v.cedula === cedula); if (vendedor) { // Archivar primero las ventas del vendedor archivarVentasVendedor(vendedor.id); // Confirmación antes de eliminar var confirmacion = confirm("¿Estás seguro de que deseas eliminar este vendedor?"); if (confirmacion) { // Eliminar vendedor vendedores = vendedores.filter(v => v.cedula !== cedula); localStorage.setItem('Vendedores', JSON.stringify(vendedores)); alert('Vendedor eliminado correctamente.'); } } else { alert('Vendedor no encontrado.'); } }
